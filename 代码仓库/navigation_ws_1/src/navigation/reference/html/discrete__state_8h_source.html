129  double getDiagonalDistance(DiscreteState other)
130  {
131  int num_x = abs(int(x_i - other.x_i));
132  int num_y = abs(int(y_i - other.y_i));
133  int max = std::max(num_x, num_y);
134  int min = std::min(num_x, num_y);
135 
136  return max - min + sqrt(2) * min;
137  }
