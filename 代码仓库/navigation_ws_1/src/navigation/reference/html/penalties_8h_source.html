57 inline double penaltyBoundToInterval(const double& var,const double& a,const double& epsilon)
58 {
59  if (var < -a+epsilon)
60  {
61  return (-var - (a - epsilon));
62  }
63  if (var <= a-epsilon)
64  {
65  return 0.;
66  }
67  else
68  {
69  return (var - (a - epsilon));
70  }
