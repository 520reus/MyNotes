70  DynamicCostmap(costmap_2d::Costmap2DROS *static_map, int max_layers,
71  ros::Duration time_resolution, std::string dynamic_layers_plugin) :
72  static_map_(static_map),
73  time_resolution_(time_resolution),
74  dl_loader_("lattice_planner", "lattice_planner::DynamicLayers")
75 
76  {
77  //load the plugin for the dynamic layers
78  try
79  {
80  ROS_INFO("dynamic_costmap: Using dynamic layers plugin %s",
81  dynamic_layers_plugin.c_str());
82 
83  dynamic_layers_ =
84  dl_loader_.createInstance(dynamic_layers_plugin);
85 
86  dynamic_layers_->initialize(static_map, max_layers, time_resolution);
87  }
88 
89  catch(pluginlib::PluginlibException& ex)

91  ROS_ERROR("dynamic_costmap: failed to load dynamic layers plugin: %s", ex.what());
92  }
93  }
