40  void computeError()
41  {
42  ROS_ASSERT_MSG(cfg_ && _measurement && robot_model_, "You must call setTebConfig(), setObstacle() and setRobotModel() on EdgeVelocityObstacleRatio()");
43  const VertexPose* conf1 = static_cast<const VertexPose*>(_vertices[0]);
44  const VertexPose* conf2 = static_cast<const VertexPose*>(_vertices[1]);
45  const VertexTimeDiff* deltaT = static_cast<const VertexTimeDiff*>(_vertices[2]);
46 
47  const Eigen::Vector2d deltaS = conf2->estimate().position() - conf1->estimate().position();
48 
49  double dist = deltaS.norm();
50  const double angle_diff = g2o::normalize_theta(conf2->theta() - conf1->theta());
51  if (cfg_->trajectory.exact_arc_length && angle_diff != 0)
52  {
53  double radius = dist/(2*sin(angle_diff/2));
54  dist = fabs( angle_diff * radius ); // actual arg length!
55  }
56  double vel = dist / deltaT->estimate();
57 
58  vel *= fast_sigmoid( 100 * (deltaS.x()*cos(conf1->theta()) + deltaS.y()*sin(conf1->theta())) ); // consider direction
59 
60  const double omega = angle_diff / deltaT->estimate();
61 
62  double dist_to_obstacle = robot_model_->calculateDistance(conf1->pose(), _measurement);
63 
64  double ratio;
65  if (dist_to_obstacle < cfg_->obstacles.obstacle_proximity_lower_bound)
66  ratio = 0;
67  else if (dist_to_obstacle > cfg_->obstacles.obstacle_proximity_upper_bound)
68  ratio = 1;
69  else
70  ratio = (dist_to_obstacle - cfg_->obstacles.obstacle_proximity_lower_bound) /
71  (cfg_->obstacles.obstacle_proximity_upper_bound - cfg_->obstacles.obstacle_proximity_lower_bound);
72  ratio *= cfg_->obstacles.obstacle_proximity_ratio_max_vel;
73 
74  const double max_vel_fwd = ratio * cfg_->robot.max_vel_x;
75  const double max_omega = ratio * cfg_->robot.max_vel_theta;
76  _error[0] = penaltyBoundToInterval(vel, max_vel_fwd, 0);
77  _error[1] = penaltyBoundToInterval(omega, max_omega, 0);
78 
79  ROS_ASSERT_MSG(std::isfinite(_error[0]) || std::isfinite(_error[1]), "EdgeVelocityObstacleRatio::computeError() _error[0]=%f , _error[1]=%f\n",_error[0],_error[1]);
80  }
