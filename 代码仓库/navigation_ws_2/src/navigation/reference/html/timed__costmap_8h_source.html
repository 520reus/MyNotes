171  void publishCostmap()
172  {
173  //wrap the cost map in OccupancyGrid form for rviz
174  nav_msgs::OccupancyGrid one_costmap;
175  one_costmap.header.frame_id = frame_id;
176  one_costmap.header.stamp = time;
177  one_costmap.info.width = size_x;
178  one_costmap.info.height = size_y;
179  one_costmap.info.origin.position.x = origin_x;
180  one_costmap.info.origin.position.y = origin_y;
181  one_costmap.info.origin.position.z = 0.0;
182  one_costmap.info.resolution = resolution;
183 
184  //scale the entries with a factor and fill the OccupancyGrid
185  double scale_factor = 100;
186 
187  for(int j=0; j<one_costmap.info.height * one_costmap.info.width; j++)
188  one_costmap.data.push_back(getCost(j) * scale_factor/costmap_2d::LETHAL_OBSTACLE);
189 
190  //publish
191  costmap_pub_.publish(one_costmap);
192  }

198  void publishCostCloud()
199  {
200  //create new point cloud object
201  pcl::PointCloud<pcl::PointXYZRGB>::Ptr cost_cloud(new pcl::PointCloud<pcl::PointXYZRGB>);
202 
203  double max_height = 1.5; //max height of costmap in meters
204 
205  //fill the point cloud
206  for(int j=0; j<size_x * size_y; j++)
207  {
208  int mx, my;
209  indexToCells(j, mx, my);
210  pcl::PointXYZRGB pnt;
211  pnt.x = mx * resolution + origin_x;
212  pnt.y = my * resolution + origin_y;
213  //height of the point is determined by the cost value of the cell
214  pnt.z = max_height * ((double) getCost(j) / (double) costmap_2d::LETHAL_OBSTACLE);
215  double z_normalized = ((double) getCost(j) / (double) costmap_2d::LETHAL_OBSTACLE);
216  //color of the point is also determined by the cost value of the cell
217  pnt.r = 255 * (z_normalized);
218  pnt.b = 255 * (1 - z_normalized);
219  pnt.g = 255 * (1 - (fabs(z_normalized - 0.5) * 2));
220 
221  if(z_normalized == 0)
222  {
223  pnt.z = -1; //move under ground plane for viz
224  pnt.r = 255;
225  pnt.g = 255;
226  pnt.b = 255;
227  }
228 
229  cost_cloud->push_back(pnt);
230  }
231 
232  //specify some general properties of the point cloud
233  cost_cloud->is_dense = false;
234 
235  std_msgs::Header header;
236  header.frame_id = frame_id;
237  header.stamp = time;
238  cost_cloud->width = size_x;
239  cost_cloud->height = size_y;
240  cost_cloud->header = pcl_conversions::toPCL(header);
241 
242  //publish
243  cost_cloud_pub_.publish(*cost_cloud);
244  }

250  void saveAsMapImage(std::string path)
251  {
252  //wrap the costmap in costmap_2d::Costmap2D form
253  costmap_2d::Costmap2D map(size_x, size_y, resolution, origin_x, origin_y);
254  for(unsigned int i = 0; i<size_x * size_y; i++)
255  {
256  //fill the map
257  int idx_x, idx_y;
258  indexToCells(i, idx_x, idx_y);
259  map.setCost(idx_x,idx_y,getCost(i));
260  }
261  //save map as image
262  map.saveMap(path);
263  }
